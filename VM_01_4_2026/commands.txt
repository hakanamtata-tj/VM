================Check Pnl by Keys without hedge================================
SELECT 
    `key`,
    SUM(pnl) AS total_pnl_sum,
    MIN(entry_time) AS first_entry_time,
    MAX(exit_time) AS last_exit_time
FROM completed_trades
GROUP BY `key`;


================Check total_pnl by Keys with hedge================================
SELECT 
    `key`,
    SUM(total_pnl) AS total_pnl_sum,
    MIN(entry_time) AS first_entry_time,
    MAX(exit_time) AS last_exit_time
FROM completed_trades
GROUP BY `key`;

Query to get pnl prod

select sum(total_pnl),sum(total_pnl*75),(sum(total_pnl*75)*100)/190000 ,STRFTIME('%m', exit_time),key,user_id,real_trade from completed_trades 
where user_id = 3 and STRFTIME('%m', exit_time) = '01' and real_trade = 'yes'
group by user_id,key,STRFTIME('%m', exit_time),real_trade;
